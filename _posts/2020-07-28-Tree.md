#### 구현
~~~ c++
class Node{
public:
  int value;
  Node* left;
  Node* right;

  Node(){this->value = 0; this->left=null; this->right=null;} 
  Node(int _value){this->value = _value; this->left=null; this->right=null;}
  Node(int _value, Node* _left, Node* _right){this->value = _value; this->left=_left; this->right=_right;}
}
~~~

#### 순회    
##### inorder, 전위    
> now -> left -> right    
~~~ c++
void inorder(Node* now){
  if(now != null){
    now->value;
    inorder(T->left);
    inorder(T->right);
  }
}
~~~
##### postorder, 중위
> left -> now -> right
~~~ c++
void postorder(Node* now){
  if(now != null){
    postorder(now->left);
    now->value;
    postorder(now->right);
  }

}
~~~
##### preorder, 후위
> left -> right -> now
~~~ c++
void preorder(Node* now){
  if(now != null){
    preorder(now->left);
    preorder(now->right);
    now->value;
  }
}
~~~
